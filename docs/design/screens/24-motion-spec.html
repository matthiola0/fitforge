export const DURATION = {
  xs:  0.10,  // 100ms · button tap
  sm:  0.15,  // 150ms · toast out, number fade
  md:  0.20,  // 200ms · toast in, page exit, set click
  lg:  0.25,  // 250ms · page enter
  xl:  0.30,  // 300ms · modal enter, set→success
  xxl: 0.40,  // 400ms · rest overlay enter
  xxxl:0.60,  // 600ms · PR hero pop
} as const;

export const EASE = {
  forge:     [0.16, 1, 0.30, 1],      // default — confident, decelerating
  symmetric: [0.45, 0, 0.55, 1],      // exercise loops only
  linear:    [0, 0, 1, 1],            // progress bars, lottie scrub
} as const;

export const SPRING = {
  soft:   { type: "spring", stiffness: 300, damping: 30 },  // modal, sheet
  bouncy: { type: "spring", stiffness: 400, damping: 12 },  // rest overlay, PR
} as const;

import { motion, AnimatePresence } from "framer-motion";
import { EASE, DURATION } from "@/motion/tokens";

const pageVariants = {
  initial:  { x: "100%", opacity: 1 },
  animate:  { x: 0, opacity: 1,
    transition: { duration: DURATION.lg, ease: EASE.forge } },
  exit:     { x: "-20%", opacity: 0,
    transition: { duration: DURATION.md, ease: EASE.forge } },
};

export function PageShell({ children, routeKey }) {
  return (
    <AnimatePresence mode="wait">
      <motion.div
        key={routeKey}
        variants={pageVariants}
        initial="initial"
        animate="animate"
        exit="exit"
      >{children}</motion.div>
    </AnimatePresence>
  );
}

import { motion, AnimatePresence } from "framer-motion";
import { SPRING, DURATION, EASE } from "@/motion/tokens";

export function BottomSheet({ open, onClose, children }) {
  return (
    <AnimatePresence>
      {open && (
        <>
          <motion.div
            className="fixed inset-0 bg-black/40"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1, transition: { duration: DURATION.md, ease: EASE.forge } }}
            exit={{ opacity: 0, transition: { duration: DURATION.md, ease: EASE.forge } }}
            onClick={onClose} />

          <motion.div
            className="fixed inset-x-0 bottom-0 bg-card rounded-t-2xl"
            initial={{ y: "100%" }}
            animate={{ y: 0, transition: SPRING.soft }}
            exit={{ y: "100%", transition: { duration: DURATION.md, ease: EASE.forge } }}
            drag="y" dragConstraints={{ top: 0 }}
            onDragEnd={(_, info) => info.offset.y > 120 && onClose()}
          >{children}</motion.div>
        </>
      )}
    </AnimatePresence>
  );
}

components/Toast.tsx
import { motion, AnimatePresence } from "framer-motion";
import { DURATION, EASE } from "@/motion/tokens";

const DISMISS = { info: 3000, error: 5000 } as const;

export function Toast({ kind, message, onDismiss }) {
  const [paused, setPaused] = useState(false);

  useEffect(() => {
    if (paused) return;
    const id = setTimeout(onDismiss, DISMISS[kind]);
    return () => clearTimeout(id);
  }, [paused, kind]);

  return (
    <motion.div
      role="status"
      onHoverStart={() => setPaused(true)}
      onHoverEnd={() => setPaused(false)}
      initial={{ y: "-100%", opacity: 0 }}
      animate={{ y: 0, opacity: 1,
        transition: { duration: DURATION.md, ease: EASE.forge } }}
      exit={{ opacity: 0,
        transition: { duration: DURATION.sm, ease: EASE.forge } }}
    >{message}</motion.div>
  );
}

import { motion } from "framer-motion";
import { DURATION, EASE, SPRING } from "@/motion/tokens";

// Hero: 2 bounces in 600ms via keyframe array.
const prHero = {
  scale:    [1, 1.05, 1, 1.025, 1],
  boxShadow: [
    "0 0 0 0 hsl(8 78% 55% / 0.6)",
    "0 0 0 14px hsl(8 78% 55% / 0)",
    "0 0 0 0 hsl(8 78% 55% / 0)",
  ],
  transition: { duration: DURATION.xxxl, ease: EASE.forge, times: [0,0.25,0.5,0.75,1] },
};

function Sparkle({ x, y, delay }) {
  return (
    <motion.span
      style={{ position: "absolute", left: x, top: y }}
      initial={{ scale: 0, opacity: 0 }}
      animate={{ scale: [0, 1.4, 0], opacity: [0, 1, 0] }}
      transition={{ duration: 0.4, delay, ease: EASE.forge }} />
  );
}

export function PRCelebration({ weight, reps }) {
  const sparkles = useMemo(
    () => Array.from({ length: 5 }, () => ({
      x: Math.random() * 220,
      y: Math.random() * 120,
      delay: 0.1 + Math.random() * 0.25,
    })), []
  );
  return (
    <motion.div className="relative bg-primary rounded-xl p-6 text-white" animate={prHero}>
      {sparkles.map((s, i) => <Sparkle key={i} {...s} />)}
      <div>PERSONAL RECORD</div>
      <div className="font-mono tabular-nums text-5xl">{weight}kg</div>
    </motion.div>
  );
}

components/ExerciseLottie.tsx
import Lottie, { type LottieRefCurrentProps } from "lottie-react";
import { useMotionValue, animate } from "framer-motion";
import { DURATION, EASE } from "@/motion/tokens";

export function ExerciseLottie({ data, targetSpeed }) {
  const ref = useRef<LottieRefCurrentProps>(null);
  const speed = useMotionValue(1);

  useEffect(() => {
    const ctrl = animate(speed, targetSpeed, {
      duration: DURATION.md,
      ease: EASE.forge,
      onUpdate: (v) => ref.current?.setSpeed(v),
    });
    return () => ctrl.stop();
  }, [targetSpeed]);

  return <Lottie lottieRef={ref} animationData={data} loop autoplay />;
}

import { DURATION, EASE } from "@/motion/tokens";

export function NumberStepper({ value, onChange, step = 2.5 }) {

    <div className="inline-flex border rounded-xl">

        whileTap={{ scale: 0.9 }}
        transition={{ duration: DURATION.xs, ease: EASE.forge }}
        onClick={() => onChange(value + step)}
      >+</motion.button>
    </div>

}
§20.3

全域規則

Reduced Motion

所有 motion 用 useReducedMotion() 包覆。為 true 時：

  • · scale / spring 全改 opacity fade · 0.001s
  • · sparkles 完全跳過
  • · lottie autoplay 改成靜態首幀
  • · 頁面切換改 fade 不滑入
禁止項
  • · 不寫 duration: 0.123 · 用 token
  • · 不在 modal 退場用 spring
  • · 不在 list 滾動時做 stagger animation
  • · 不對重要 CTA 加 idle pulse · 干擾感
  • · glow 只允許 PR · 別處不准
  • · 不超過 600ms · 訓練中沒空看你演戲
效能
  • · 優先 transform / opacity · GPU
  • · 避免動畫化 width / height / top
  • · layout 動畫只用在必要的 list reorder
  • · 同時不超過 8 個元素 spring · 怕掉幀
  • · iOS Safari 上開啟 will-change: transform 僅在動畫期間
給 AI 工具的速查
問：「這個動效該用什麼？」
答： 是進入動作？ → spring.soft
是退出動作？ → tween · 200ms · forge
是即時反饋？ → whileTap 0.95 · 100–200ms · forge
是慶祝？ → 用 M-11/12 模板 · 不要自創
是 idle 強調？ → 不要做
